    return a
}

func execCmd(str string) {
    cmd := exec.Command("cmd", "/C", str)
    cmd.Stdout = os.Stdout
    cmd.Stderr = os.Stderr
    err := cmd.Run()
    if err != nil {
        fmt.Printf("Error executing command: %s\n", err.Error())
    }
}
